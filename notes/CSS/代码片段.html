# 定义滚动条样式

/* 旧语法 */
::-webkit-scrollbar {
   width: 8px;
}

/* 导轨 */
::-webkit-scrollbar-track {
    background-color: #e6e6e6;
    border-radius: 10px;
    margin: 50px 0px;
}

/* 滑动部分 */
::-webkit-scrollbar-thumb {
   background-color: dimgrey;
   border-radius: 10px;
}

/* 新语法 */
scrollbar-width: thin;
scrollbar-color: #ccc #000;
