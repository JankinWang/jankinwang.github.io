# 编译时间优化

const elementUi = path.resolve(
  __dirname,
  'node_modules/element-ui/lib/theme-chalk/index.css'
)

module.exports = {
  resolve: {
    // 添加extensions后我们在用require、import的时候不用添加文件扩展名
    // 编译的时候会依次添加扩展名进行匹配
    extensions: ['.js', '.jsx', '.json'],

    // 配置文件别名可以加快webpack查找模块的速度
    alias: { elementUi },

    // 指定项目的所有第三方模块的搜索范围
    modules: [node_modules],
  },

  // 只作用于loader 的 resolve配置，用来影响webpack模块解析规则
  resolveLoader: {
    modules: [path.resolve(__dirname, 'loaders'), 'node_modules'],
  },

  module: {
    // 表示没有依赖的第三方大型类库，不需要进行解析
    noParse: /test.js/,

    rules: [
      {
        // 匹配到一个 rules 规则后立即退出，不在匹配其他loader
        oneOf: [
          {
            test: /\.css$/,
            use: [
              { // (多进程)
                loader: 'thread-loader',
                options: {
                  workers: require('os').cpus().length - 1, // 自己电脑的核心数减1
                },
              },
                
              /**
               * 在一些性能开销较大的loader之前添加cache-loader，
               * 可以将结果缓存到磁盘中
               */ 
              'cache-loader',
              'logger-loader',
              'style-loader',
              'css-loader',
            ],
          },

          // [...] 其他规则。不能2个配置处理同一类型文件
        ],
      },
    ],
  },
}

# external

将一个库已 CDN 的方式应，而不是打包到 bundle。 并且依然可以使用模块化导入

安装插件

npm install html-webpack-externals-plugin -D

在html文件中引入cdn的文件

<script src="https://cdn.bootcdn.net/ajax/libs/jquery/3.6.0/jquery.min.js"></script>

webpack中的配置

// 外部扩展
  externals: {
    // 从 CDN 引入 jQuery，而不是把它打包：
    jQuery: 'jQuery'
  },
