# 虚拟DOM和Diff算法

# 虚拟DOM(vnode)

# 概念

虚拟 DOM 就是用 JavaScript 对象去模拟 DOM 结构

# 尝试用 Js 表达一个 DOM 结构

HTML 结构

<div id="app" class="container">
    <h1>虚拟DOM</h1>
    <ul style="color: orange;">
        <li>第一项</li>
        <li>第二项</li>
        <li>第三项</li>
    </ul>
</div>

# Snabbdom 虚拟DOM库

github (opens new window)

# vnode 对象结构

{ 
    sel,      // dom 选择器, 如: div#id.class
    data,     // dom 元素的属性、class、style 等
    children, // 子元素组成的数组
    text,     // 元素的文本内容, children 与 text 同时只会用一个
    elm,      // 绑定的真实 DOM，是虚拟DOM 对应的真实DOM
    key 	  // key Diff 时用来判断是否为相同的节点
}

# Example

import {
  init,
  classModule,
  propsModule,
  styleModule,
  eventListenersModule,
  h,
} from "snabbdom";

const patch = init([
  // Init patch function with chosen modules
  classModule, // makes it easy to toggle classes
  propsModule, // for setting properties on DOM elements
  styleModule, // handles styling on elements with support for animations
  eventListenersModule, // attaches event listeners
]);

const container = document.getElementById("container");

const vnode = h("div#container.two.classes", { on: { click: someFn } }, [
  h("span", { style: { fontWeight: "bold" } }, "This is bold"),
  " and this is just normal text",
  h("a", { props: { href: "/foo" } }, "I'll take you places!"),
]);
// Patch into empty DOM element – this modifies the DOM as a side effect
patch(container, vnode);

const newVnode = h(
  "div#container.two.classes",
  { on: { click: anotherEventHandler } },
  [
    h(
      "span",
      { style: { fontWeight: "normal", fontStyle: "italic" } },
      "This is now italic type"
    ),
    " and this is still just normal text",
    h("a", { props: { href: "/bar" } }, "I'll take you places!"),
  ]
);

// Second `patch` invocation
patch(vnode, newVnode); // Snabbdom efficiently updates the old view to the new state

# patch 函数

const container = document.getElementById('container');

// vnode -> 空容器
patch(container, vnode);

patch 接收两个参数： oldVnode、vnode

// patch 函数定义
function patch(oldVnode: VNode|Element, vnode: VNode): Vnode
  1. 如果 oldVnode 是 VNode 还是 Element
    1. VNode : 进行下一步
    2. Element: 创建一个空的 vnode 来作为 oldVnode，并将此Element 绑定到空 vnode.elm, 用来作为更新节点是的目标.
  2. 比较 oldVnode 与 vnode 是否为同一个节点( 标签名 和 key 相同):
    1. 是: 用 patchVnode 函数更新 vnode
    2. 否: 用 vnode 创建真实的DOM元素来替换 oldVnode 对应的DOM元素
  3. patchVnode
    1. vnode.elm = oldVnode.elm 设置 vnode 关联的DOM元素
    2. vnode 有text 无 children 且 text 有修改：
      • 删除旧的 children（有的话）
      • 设置 text
    3. vnode无 text 有 children
      1. 新旧 vnode 都有 children， 用 updateChildren 函数更新 children
      2. 新 vnode 有children, 旧没有: 添加 children 到 DOM
      3. 旧 vnode 有children, 新没有: 删除 DOM 的children
      4. 旧 vnode 有 text: 清空DOM 的 text
    4. updateChildren

# patch 函数源码

return function patch(
    oldVnode: VNode | Element | DocumentFragment,
    vnode: VNode
  ): VNode {
    let i: number, elm: Node, parent: Node;
    const insertedVnodeQueue: VNodeQueue = [];
    for (i = 0; i < cbs.pre.length; ++i) cbs.pre[i]();

    if (isElement(api, oldVnode)) {
      oldVnode = emptyNodeAt(oldVnode);
    } else if (isDocumentFragment(api, oldVnode)) {
      oldVnode = emptyDocumentFragmentAt(oldVnode);
    }

    if (sameVnode(oldVnode, vnode)) {
      patchVnode(oldVnode, vnode, insertedVnodeQueue);
    } else {
      elm = oldVnode.elm!;
      parent = api.parentNode(elm) as Node;

      createElm(vnode, insertedVnodeQueue);

      if (parent !== null) {
        api.insertBefore(parent, vnode.elm!, api.nextSibling(elm));
        removeVnodes(parent, [oldVnode], 0, 0);
      }
    }

    for (i = 0; i < insertedVnodeQueue.length; ++i) {
      insertedVnodeQueue[i].data!.hook!.insert!(insertedVnodeQueue[i]);
    }
    for (i = 0; i < cbs.post.length; ++i) cbs.post[i]();
    return vnode;
  };
}

# Diff 算法

新旧虚拟 dom 比较:

首先会比较 标签名 和 key 是否相同

  • 标签名不同。直接删除，不继续深度比较。

  • 标签名 和 key 都相同。则认为是相同的节点，不再继续深度比较。

  • 标签名相同，key 不同。
